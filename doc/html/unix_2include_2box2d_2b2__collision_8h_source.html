283 inline bool b2AABB::IsValid() const
284 {
285  b2Vec2 d = upperBound - lowerBound;
286  bool valid = d.x >= 0.0f && d.y >= 0.0f;
287  valid = valid && lowerBound.IsValid() && upperBound.IsValid();
288  return valid;
289 }

291 inline bool b2TestOverlap(const b2AABB& a, const b2AABB& b)
292 {
293  b2Vec2 d1, d2;
294  d1 = b.lowerBound - a.upperBound;
295  d2 = a.lowerBound - b.upperBound;
296 
297  if (d1.x > 0.0f || d1.y > 0.0f)
298  return false;
299 
300  if (d2.x > 0.0f || d2.y > 0.0f)
301  return false;
302 
303  return true;
304 }
